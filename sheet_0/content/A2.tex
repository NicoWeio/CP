\section{Aufgabe 2} \label{sec:A2}

\begin{align*}
    \frac{1}{\sqrt{x}} - \frac{1}{\sqrt{x+1}}
        &\quad\longrightarrow\quad
    \frac{\sqrt{x + 1} - \sqrt{x}}{\sqrt{x + 1} \sqrt{x}} \\
    \\
    \frac{1 - \cos(x)}{\sin(x)}
        &\quad\longrightarrow\quad
    \frac{1}{\sin(x)} - \frac{\cos(x)}{\sin(x)} \\
    \\
    \sin(x + \delta) - \sin(x)
        &\quad\longrightarrow\quad
    \sin(x) \cos(\delta) + \cos(x) \sin(\delta) - \sin(x) \\
\end{align*}
